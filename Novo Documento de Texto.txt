var slideIndex = 1;
showDivs(slideIndex);

function plusDivs(n) {
  showDivs(n);
}

function showDivs(n) {
  var i;
  var x = document.getElementsByClassName("imagemRotativo");
  var z = document.getElementsByClassName("rotativoBotao");
  for (i = 0; i < x.length; i++) {
    if (i = n) {
      x[i].classList.add("ativo");
      z[i].classList.add("ativo");
    }
    else{
    x[i].classList.remove("ativo");
    z[i].classList.remove("ativo");
    }
  }
}
